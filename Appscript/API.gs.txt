const SPREADSHEET_ID = "1VtFVWUFJOFgO_6GicMapl5g0OGeXMKUZsDRPe6-j_yw";

function getSheetSafely(name) {
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  let sheet = ss.getSheetByName(name);
  if (!sheet) {
    sheet = ss.insertSheet(name);
    sheet.appendRow(["id", "json_data"]);
  }
  return sheet;
}

/**
 * Generic data loader
 */
function loadAllData(sheetName) {
  try {
    const sheet = getSheetSafely(sheetName);
    const data = sheet.getDataRange().getValues();
    if (data.length < 2) return [];
    
    // Skip headers
    data.shift();
    
    return data.map(row => {
      try {
        return JSON.parse(row[1]);
      } catch (e) {
        console.error(`Error parsing JSON in ${sheetName}: ${e}`);
        return null;
      }
    }).filter(item => item !== null);
  } catch (e) {
    console.error(`Error loading data from ${sheetName}: ${e}`);
    return [];
  }
}

/**
 * Generic data saver/updater
 */
function saveItem(sheetName, item) {
  const sheet = getSheetSafely(sheetName);
  const data = sheet.getDataRange().getValues();
  const id = item.id.toString();
  const json = JSON.stringify(item);
  
  for (let i = 1; i < data.length; i++) {
    if (data[i][0].toString() === id) {
      sheet.getRange(i + 1, 2).setValue(json);
      return true;
    }
  }
  
  // If not found, append
  sheet.appendRow([id, json]);
  return true;
}

function getProjectStats() {
  const tasks = getTasks();
  let stats = { 
    todo: 0, 
    pending: 0, 
    development: 0, 
    testing: 0, 
    done: 0, 
    scrapyard: 0,
    overallCompletion: 0,
    staffWorkload: []
  };
  
  if (!tasks.length) return stats;
  
  let totalProg = 0;
  const workloadMap = {}; // name -> { active, pending }

  tasks.forEach(t => {
    const status = (t.status || '').toLowerCase();
    if (stats.hasOwnProperty(status)) stats[status]++;
    else if (status === 'inprogress') stats.development++;
    
    const milestones = t.milestones || [];
    const totalM = milestones.length > 0 ? milestones.length : 10;
    const prog = totalM > 0 ? ((t.completedMilestones || 0) / totalM) : 0;
    totalProg += prog;

    // Tally per-person workload for active (development) and pending statuses
    const isActive = (status === 'development' || status === 'inprogress');
    const isPending = (status === 'pending');
    if (isActive || isPending) {
      const assignees = (t.assignee || '').split(',').map(n => n.trim()).filter(n => n);
      assignees.forEach(name => {
        if (!workloadMap[name]) workloadMap[name] = { name, active: 0, pending: 0 };
        if (isActive) workloadMap[name].active++;
        if (isPending) workloadMap[name].pending++;
      });
    }
  });
  
  stats.overallCompletion = Math.round((totalProg / tasks.length) * 100);
  stats.staffWorkload = Object.values(workloadMap).sort((a, b) => (b.active + b.pending) - (a.active + a.pending));
  return stats;
}

// --- MUTATIONS ---

function updateTaskStatus(taskId, newStatus) {
  const tasks = getTasks();
  const task = tasks.find(t => t.id.toString() === taskId.toString());
  if (task) {
    task.status = newStatus;
    task.lastUpdated = new Date().getTime();
    return saveItem("Tasks", task);
  }
  return false;
}

function addTask(payload) {
  const id = new Date().getTime().toString();
  const task = {
    id: id,
    title: payload.title,
    status: 'todo',
    assignee: payload.assignee,
    description: payload.description,
    milestones: JSON.parse(payload.milestones),
    completedMilestones: 0,
    notes: [],
    startDate: payload.start,
    lastUpdated: new Date().getTime()
  };
  return saveItem("Tasks", task);
}

function addStaff(payload) {
  const id = new Date().getTime().toString();
  const staff = {
    id: id,
    name: payload.name,
    email: payload.email,
    role: payload.role
  };
  return saveItem("Staff", staff);
}

function updateStaffRole(staffEmail, newRole) {
  const staff = getStaff();
  const member = staff.find(s => s.email.toLowerCase() === staffEmail.toLowerCase());
  if (member) {
    member.role = newRole;
    if (!member.id) member.id = new Date().getTime().toString(); // Ensure ID for saving
    return saveItem("Staff", member);
  }
  return false;
}

function updateTaskMilestones(taskId, milestones, completedCount) {
  const tasks = getTasks();
  const task = tasks.find(t => t.id.toString() === taskId.toString());
  if (task) {
    task.milestones = milestones;
    task.completedMilestones = completedCount;
    task.lastUpdated = new Date().getTime();
    return saveItem("Tasks", task);
  }
  return false;
}

function addNoteToTask(taskId, noteText) {
  const tasks = getTasks();
  const task = tasks.find(t => t.id.toString() === taskId.toString());
  if (task) {
    if (!task.notes) task.notes = [];
    task.notes.push({
      text: noteText,
      date: new Date().toLocaleString()
    });
    task.lastUpdated = new Date().getTime();
    saveItem("Tasks", task);
    return task.notes;
  }
  return null;
}

function addIdea(payload) {
  const id = new Date().getTime().toString();
  const idea = {

    name: payload.name,
    description: payload.description,
    pros: payload.pros,
    cons: payload.cons,
    details: payload.details,
    date: new Date().toLocaleDateString()
  };
  return saveItem("Notebook", idea);
}

function deleteIdea(id) {
  const sheet = getSheetSafely("Notebook");
  const data = sheet.getDataRange().getValues();
  for (let i = 1; i < data.length; i++) {
    if (data[i][0].toString() === id.toString()) {
      sheet.deleteRow(i + 1);
      return true;
    }
  }
  return false;
}

function updateTaskDetails(id, newAssignee, newMilestones) {
  const tasks = loadAllData("Tasks");
  const task = tasks.find(t => t.id.toString() === id.toString());
  
  if (task) {
    task.assignee = newAssignee;
    task.milestones = newMilestones;
    task.completedMilestones = newMilestones.filter(m => m.completed).length;
    task.lastUpdated = new Date().getTime();
    
    return saveItem("Tasks", task);
  }
  return false;
}

function takeIdea(id, takerName) {
  const ideas = loadAllData("Notebook");
  const idea = ideas.find(i => i.id.toString() === id.toString());
  
  if (idea) {
    idea.taker = takerName;
    return saveItem("Notebook", idea);
  }
  return false;
}
